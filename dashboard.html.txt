    < script >
        let studentsData = [];

    // Загрузка данных
    async function loadData() {
        try {
            const res = await fetch('/get_students');
            const data = await res.json();
            studentsData = data.students;
            renderStudentsTable();
            renderLogsTable();
        } catch (err) {
            console.error("Ошибка загрузки:", err);
        }
    }

    // Отрисовка таблицы учеников
    function renderStudentsTable() {
        const list = document.getElementById('students-list');
        list.innerHTML = studentsData.map(s => `
                <tr>
                    <td class="fw-bold">${s.last_name} ${s.first_name}</td>
                    <td><span class="badge badge-${s.class} p-2">${s.class}</span></td>
                    <td><code>${s.id}</code></td>
                    <td><span class="badge ${s.active ? 'bg-success' : 'bg-danger'}">${s.active ? 'Активен' : 'Заблокирован'}</span></td>
                    <td class="text-end">
                        <button class="btn btn-sm btn-light" onclick="changeStatus('${s.id}', ${!s.active})">
                            ${s.active ? 'Блок' : 'Разблок'}
                        </button>
                    </td>
                </tr>
            `).join('');
        document.getElementById('student-count').textContent = studentsData.length;
    }

    // Смена статуса
    async function changeStatus(cardId, newStatus) {
        await fetch('/update_status', {
            method: 'POST',
            headers: {
                'Content-Type': 'application/json'
            },
            body: JSON.stringify({
                id: cardId,
                active: newStatus
            })
        });
        loadData();
    }

    // Сохранение настроек времени (Запрет)
    async function saveConfig() {
        const start = document.getElementById('ban-start').value;
        const end = document.getElementById('ban-end').value;
        await fetch('/update_config', {
            method: 'POST',
            headers: {
                'Content-Type': 'application/json'
            },
            body: JSON.stringify({
                start_hour: parseInt(start),
                end_hour: parseInt(end)
            })
        });
        alert("Настройки времени сохранены!");
    }

    // Добавление ученика
    document.getElementById('add-student-form').onsubmit = async function(e) {
        e.preventDefault();
        const student = {
            id: document.getElementById('card-uid').value,
            last_name: document.getElementById('last-name').value,
            first_name: document.getElementById('first-name').value,
            middle_name: document.getElementById('middle-name').value,
            class: document.getElementById('klass').value,
            active: true,
            history: []
        };
        const res = await fetch('/add_student', {
            method: 'POST',
            headers: {
                'Content-Type': 'application/json'
            },
            body: JSON.stringify(student)
        });
        if (res.ok) {
            this.reset();
            loadData();
        }
    };

    // Сканер
    async function scanCard() {
        const res = await fetch('/get_last_id');
        const id = await res.text();
        if (id) document.getElementById('card-uid').value = id;
        else alert("Карта не обнаружена");
    }

    // Журнал
    function renderLogsTable() {
        let allLogs = [];
        studentsData.forEach(s => {
            if (s.history) s.history.forEach(h => {
                allLogs.push({
                    name: `${s.last_name} ${s.first_name}`,
                    class: s.class,
                    time: h
                });
            });
        });
        allLogs.sort((a, b) => b.time.localeCompare(a.time));
        document.getElementById('logs-list').innerHTML = allLogs.map(l => `
                <tr>
                    <td>${l.name}</td>
                    <td>${l.class}</td>
                    <td>${l.time}</td>
                    <td>Вход выполнен</td>
                </tr>
            `).join('');
    }

    // Фильтрация
    function filterTable() {
        const dateIn = document.getElementById('filter-date').value;
        const hourIn = document.getElementById('filter-hour').value;
        const classIn = document.getElementById('filter-class').value;
        let fmtD = dateIn ? `${dateIn.split('-')[2]}.${dateIn.split('-')[1]}` : "";

        document.querySelectorAll('#logs-list tr').forEach(row => {
            const dt = row.cells[2].innerText; // "25.05.24 14:00"
            const cl = row.cells[1].innerText;
            const mD = !fmtD || dt.includes(fmtD);
            const mH = !hourIn || dt.split(' ')[1].startsWith(hourIn);
            const mC = !classIn || cl === classIn;
            row.style.display = (mD && mH && mC) ? "" : "none";
        });
    }

    async function sendCommand(action) {
        if (!confirm(`Вы уверены, что хотите выполнить действие: ${action}?`)) return;
        try {
            const response = await fetch(`${ESP_IP}/command?action=${action}`);
            if (response.ok) alert("Команда успешно отправлена");
        } catch (e) {
            alert("Ошибка связи с контроллером");
        }
    }

    // 3. Сохранение настроек временного ограничения
    function saveTimeConfig() {
        const start = document.getElementById('ban-start-time').value;
        const end = document.getElementById('ban-end-time').value;

        console.log(`Настройка ограничения: с ${start} до ${end}`);

        // Отправка на ESP
        fetch(`${ESP_IP}/set_ban?start=${start}&end=${end}`)
            .then(res => alert("Настройки времени сохранены!"))
            .catch(err => alert("Ошибка при сохранении"));
    }

    // 4. Умная фильтрация журнала (по Дате, Времени и Классу)
    function filterLogsTable() {
        const dateVal = document.getElementById('filter-date').value;
        const timeFrom = document.getElementById('filter-time-from').value;
        const timeTo = document.getElementById('filter-time-to').value;
        const classVal = document.getElementById('filter-class').value;

        const rows = document.querySelectorAll('#logs-list tr');
        let visibleCount = 0;

        rows.forEach(row => {
            const rowDateTimeText = row.cells[2].innerText; // Формат: "2023-10-25 14:30:05"
            const [rowDate, rowTime] = rowDateTimeText.split(' ');
            const rowClass = row.cells[1].innerText;

            // Логика фильтрации
            const matchDate = !dateVal || rowDate === dateVal;
            const matchClass = !classVal || rowClass === classVal;

            // Сравнение времени (просто как строки "14:00" >= "13:00")
            const matchTimeFrom = !timeFrom || rowTime >= timeFrom;
            const matchTimeTo = !timeTo || rowTime <= timeTo;

            if (matchDate && matchClass && matchTimeFrom && matchTimeTo) {
                row.style.display = '';
                visibleCount++;
            } else {
                row.style.display = 'none';
            }
        });

        document.getElementById('log-count').innerText = visibleCount;
    }

    // Пример функции добавления строки в журнал (для теста)
    function addLogEntry(fio, klass, datetime, event) {
        const tbody = document.getElementById('logs-list');
        const row = tbody.insertRow(0);
        row.className = 'new-row';
        row.innerHTML = `
        <td>${fio}</td>
        <td><span class="badge badge-${klass}">${klass}</span></td>
        <td>${datetime}</td>
        <td><span class="text-success">${event}</span></td>
    `;
        filterLogsTable();
    }
